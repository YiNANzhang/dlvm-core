IDENTIFIER : '[a-zA-Z][a-zA-Z0-9]*'
NUMBER : '[0-9]+'

program : (statement)*

statement : attribute
          | declaration

attribute : '@' attribute_name attribute_definition

attribute_name : 'type'
               | 'name'

attribute_definition : IDENTIFIER

declaration : assignment_declaration
            | recurrence_declaration

assignment_declaration : IDENTIFIER ':' role shape ('=' expression)?
